# 离散化

离散化是一种数据处理的技巧，本质上可以看成是一种**哈希**，其保证数据在哈希以后仍然保持原来的**全/偏序**关系

有些数据因为本身很大或者类型不支持，自身无法作为数组的下标来方便地处理，而影响最终结果的只有元素之间的相对大小关系时，我们可以将原来的数据按照排名来处理问题，即离散化。

## 方法一

通常原数组中会有重复的元素，一般把相同的元素离散化为相同的数据。

方法如下：

1. 创建原数组的副本。
2. 将副本中的值从小到大排序。
3. 将排序好的副本去重。
4. 查找原数组的每一个元素在副本中的位置，位置即为排名，将其作为离散化

```
memcpy(tmp, arr, sizeof arr);
sort(tmp + 1, tmp + 1 + n);
int len = unique(tmp + 1, tmp + 1 + n) - (tmp + 1);
for(int i = 1; i <= n; i ++) {
	arr[i] = lower_bound(tmp + 1, tmp + 1 + len, arr[i]) - tmp;
}
```

```c
vector<int> tmp(arr);
sort(tmp.begin(), tmp.end());
tmp.erase(unique(tmp.begin(), tmp.end()), tmp.end());
for(int i = 0; i < n; i ++) {
	arr[i]= lower_bound(tmp.begin(), tmp.end(), arr[i]) - tmp.begin();
}
```



[火烧赤壁]([P1496 火烧赤壁 - 洛谷 | 计算机科学教育新生态](https://www.luogu.com.cn/problem/P1496#submit))

```c
#include <bits/stdc++.h>
#define int long long
using namespace std;
const int N = 2e4 + 10;

int n, ans;
int a[N], b[N], c[2*N], st[2*N];

void solve() {
    int k = 1;
    cin >> n;
    for(int i = 1; i <= n; i ++) {
        cin >> a[i] >> b[i];
        c[k ++] = a[i];
        c[k ++] = b[i];
    }
    sort(c + 1, c + 1 + 2*n);
    for(int i = 1; i <= n; i ++) {
        a[i] = lower_bound(c + 1, c + 1 + 2*n, a[i]) - c;
        b[i] = lower_bound(c + 1, c + 1 + 2*n, b[i]) - c;
        for(int j = a[i]; j < b[i]; j ++) st[j] = 1;
    }
    for(int i = 1; i < 2*n; i ++) {
        if(st[i]) {
            ans += c[i + 1] - c[i];
        }
    }
    cout << ans << endl;
}

signed main() {
    ios::sync_with_stdio(false);
    cin.tie(0);
    cout.tie(0);
    
    solve();
    return 0;
}
```

[天际线]([P1904 天际线 - 洛谷 | 计算机科学教育新生态](https://www.luogu.com.cn/problem/P1904#submit))

```
#include <bits/stdc++.h>
#define int long long
using namespace std;
const int N = 5000 + 10;

int a[N], h[N], b[N];
int c[2*N], st[2*N], ans[2*N];

void solve() {
    int k = 0, cnt = 0;
    while(scanf("%lld%lld%lld", &a[cnt], &h[cnt], &b[cnt]) != EOF) {
        c[k ++] = a[cnt];
        c[k ++] = b[cnt];
        cnt ++;
    }
    sort(c, c + k);
    for(int i = 0; i < cnt; i ++) {
        a[i] = lower_bound(c, c + k, a[i]) - c;
        b[i] = lower_bound(c, c + k, b[i]) - c;
        for(int j = a[i]; j < b[i]; j ++) {
            if(st[j] < h[i]) {
                st[j] = h[i];
            }
        }
    }
    ans[0] = c[0];
    int x = 1;
    for(int i = 1; i < k; i ++) {
        if(st[i] != st[i - 1]) {
            ans[x ++] = st[i - 1];
            ans[x ++] = c[i];
        }
    }
    for(int i = 0; i < x; i ++) {
        printf("%lld ", ans[i]);
    }
    printf("0\n");
}

signed main() {
    ios::sync_with_stdio(false);
    cin.tie(0);
    cout.tie(0);
    
    solve();
    return 0;
}
```

二维离散化

[Overplanting S]([P1884 [USACO12FEB\] Overplanting S - 洛谷](https://www.luogu.com.cn/problem/P1884))

```
#include <bits/stdc++.h>
#define int long long
using namespace std;
const int N = 1010;

struct node {
    int x1, y1, x2, y2;
}rect[N];

int n, ans;
int a[2*N], b[2*N], c[2*N][2*N];

void solve() {
    int x1, y1, x2, y2;
    cin >> n;
	for(int i = 0; i < n; i ++) {
        cin >> x1 >> y1 >> x2 >> y2;
        rect[i].x1 = y2 , a[i*2] = y2;
        rect[i].x2 = y1 , a[i*2 + 1] = y1;
        rect[i].y1 = x1 , b[i*2] = x1;
        rect[i].y2 = x2 , b[i*2 + 1] = x2;
    }

    sort(a, a + 2*n);
    sort(b, b + 2*n);
    int lena = unique(a, a + 2*n) - a;
    int lenb = unique(b, b + 2*n) - b;

    for(int i = 0; i < n; i ++) {
        rect[i].x1 = lower_bound(a, a + lena, rect[i].x1) - a;
        rect[i].x2 = lower_bound(a, a + lena, rect[i].x2) - a;
        rect[i].y1 = lower_bound(b, b + lenb, rect[i].y1) - b;
        rect[i].y2 = lower_bound(b, b + lenb, rect[i].y2) - b;
        for(int j = rect[i].x1; j < rect[i].x2; j ++) {
            for(int k = rect[i].y1; k < rect[i].y2; k ++) {
                c[j][k] = 1;
            }
        }
    }

    for(int i = 0; i < lena - 1; i ++) {
        for(int j = 0; j < lenb - 1; j ++) {
            if(c[i][j]) {
                ans += (a[i + 1] - a[i]) * (b[j + 1] - b[j]);
            }
        }
    }
    cout << ans << endl;
}

signed main() {
    ios::sync_with_stdio(false);
    cin.tie(0);
    cout.tie(0);
    
    solve();
    return 0;
}
```



## 方法二

根据题目要求，有时候会把相同的元素根据输入顺序离散化为不同的数据

方法如下：

1. 创建原数组的副本，同时记录每个元素出现的位置。
2. 将副本按值从小到大排序，当值相同时，按出现顺序从小到大排序。
3. 将离散化后的数字放回原数组。



[Cow Lineup S]([P3029 [USACO11NOV\] Cow Lineup S - 洛谷](https://www.luogu.com.cn/problem/P3029))

```
#include <bits/stdc++.h>
#define int long long
using namespace std;
const int N = 50010;

struct node {
    int x, y;
}a[N];

map<int, int> m;
int n, k, h, ans = 1e18;

bool cmp(node a, node b) {
    return a.x < b.x;
}

void solve() {
    cin >> n;
    for(int i = 0; i < n; i ++) {
        cin >> a[i].x >> a[i].y;
        m[a[i].y] = 1;
    }
    int num = m.size();
    sort(a, a + n, cmp);

    for(int i = 0; i < n; i ++) {
        m[a[i].y] ++;
        if(m[a[i].y] == 2) {    
            k ++;
            while(k == num) {
                ans = min(ans, a[i].x - a[h].x);
                m[a[h].y] --;
                if(m[a[h].y] == 1) {
                    k --;
                }
                h ++;
            }
        }
    }
    cout << ans << endl;
}

signed main() {
    ios::sync_with_stdio(false);
    cin.tie(0);
    cout.tie(0);
    
    solve();
    return 0;
}
```

[并查集＋离散化]([P1955 [NOI2015\] 程序自动分析 - 洛谷](https://www.luogu.com.cn/problem/P1955))

```
#include <bits/stdc++.h>
#define int long long
using namespace std;
const int N = 1e6 + 10;

int t;
vector<int> pa;
map<int, int> m;

struct node {
    int x, y;
}na[N], nb[N];

inline int read() {
    int x = 0, f = 1;
    char ch = getchar();
    while (ch < '0' || ch > '9') {
        if (ch == '-') f = -1;
        ch = getchar();
    }
    while (ch >= '0' && ch <= '9') {
        x = x * 10 + ch - '0';
        ch = getchar();
    }
    return x * f;
}

int unfind(int x) {
    if(pa[x] != x) {
        pa[x] = unfind(pa[x]);
        return pa[x];
    }
    return x;
}

void unionset(int x, int y) {
    int rx = unfind(x);
    int ry = unfind(y);
    if(rx != ry) {
        pa[rx] = ry;
    }
}

void solve() {
    t = read();
    while(t --) {
        pa.resize(N), m.clear();
        int n = read(), k = 1;
        int f, ia, ib;
        f = ia = ib = 0;
        for(int i = 1; i <= N; i ++) pa[i] = i;
        for(int i = 0; i < n; i ++) {
            int x = read(), y = read(), e = read();
            if(!m.count(x)) m[x] = k ++;
            if(!m.count(y)) m[y] = k ++;
            if(e == 1) {
                na[ia].x = x, na[ia ++].y = y;
            }
            else {
                nb[ib].x = x, nb[ib ++].y = y;
            }
        }
        for(int i = 0; i < ia; i ++) {
            unionset(m[na[i].x], m[na[i].y]);
        }
        for(int i = 0; i < ib; i ++) {
            int x = m[nb[i].x], y = m[nb[i].y];
            int rx = unfind(x);
            int ry = unfind(y);
            if(rx == ry) {
                f = 1;
                break;
            }
        }

        if(f) printf("NO\n");
        else printf("YES\n"); 
    }
}

signed main() {
    ios::sync_with_stdio(false);
    cin.tie(0);
    cout.tie(0);
    
    solve();
    return 0;
}
```
