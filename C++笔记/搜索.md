## DFS

[单词接龙]([P1019 [NOIP2000 提高组\] 单词接龙 - 洛谷 | 计算机科学教育新生态](https://www.luogu.com.cn/problem/P1019))

```c
#include <bits/stdc++.h>
#define int long long
using namespace std;
const int N = 30;

int n, res = 0;
string word[N];
int used[N], g[N][N]; // 存第i个单词与第j个单词重合长度

void dfs(string dragon, int x) {
	res = max(res, (int)dragon.size());
	used[x] ++;
	for(int i = 0; i < n; i ++) {
		if(g[x][i] && used[i] < 2) {
			dfs(dragon + word[i].substr(g[x][i]), i);
		}
	}
	used[x] --;
}

void solve() {
	char start;
	cin >> n;
	for(int i = 0; i < n; i ++) cin >> word[i];
	cin >> start;
	
	for(int i = 0; i < n; i ++) {
		for(int j = 0; j < n; j ++) {
			string a = word[i], b = word[j];
			for(int k = 1; k < min(a.size(), b.size()); k ++) {
				if(a.substr(a.size() - k, k) == b.substr(0, k)) {
					g[i][j] = k;
					break;
				}
			}
		}
	}
	for(int i = 0; i < n; i ++) {
		if(word[i][0] == start) {
			dfs(word[i], i);
		}
	}
	cout << res << endl;
}

signed main() {
    ios::sync_with_stdio(false);
    cin.tie(0);
    cout.tie(0);
    
    solve();
    return 0;
}
```

```c
#include <bits/stdc++.h>
#define int long long
using namespace std;
const int N = 25;

int n, tot, r;
int path[N], ans[N], nums[N];
bool g[N][N], st[N];

bool check(int x) {
	for(int i = 1; i <= n; i ++) {
		if(g[x][i] && !st[i]) { //如果联通且没走过
			return true;
		}
	}
	return false;
}

void dfs(int x, int tmp, int cnt) {
	if(!check(x)) {
		if(cnt > tot) {
			tot = cnt;
			r = tmp;
			for(int i = 1; i <= tmp; i ++) {
				ans[i] = path[i];
			}
		}
		return ;
	}
	for(int i = 1; i <= n; i ++) {
		if(g[x][i] && !st[i]) {
			st[x] = true;
			path[tmp + 1]= i;
			dfs(i, tmp + 1, cnt + nums[i]);
			st[x] = false;
		}
	}
}

void solve() {
    cin >> n;
	for(int i = 1; i <= n; i ++) cin >> nums[i];
	for(int i = 1; i < n; i ++) {
		for(int j = i + 1; j <= n; j ++) {
			cin >> g[i][j];
		}
	}
	
	for(int i = 1; i <= n; i ++) {
		st[i] = true;
		path[1] = i;
		dfs(i, 1, nums[i]);
		st[i] = false;
	}
	
	for(int i = 1; i <= r; i ++) {
		cout << ans[i] << " ";
	}
	cout << endl << tot;
}

signed main() {
    ios::sync_with_stdio(false);
    cin.tie(0);
    cout.tie(0);
    
    solve();
    return 0;
}
```

```
int n, m;

        cin >> a[i].t >> a[i].f >> a[i].h;
```

```c
#include <bits/stdc++.h>
#define int long long
using namespace std;
const int N = 1e6;

int n, q, cnt;
vector<int> tree[N];
int in[N], out[N], bel[N];

void dfs(int x) {
    in[x] = ++ cnt;
    bel[cnt] = x;
    for(auto &i : tree[x]) {
        dfs(i);
    } 
    out[x] = cnt;
}

void solve() {
	cin >> n >> q;
    for(int i = 2; i <= n; i ++) {

    for(int i = 1; i <= q; i ++) {
        int u, k;
        cin >> u >> k;
        int qry = in[u] + k - 1;
        if(qry > out[u]) {
            cout << -1 << "\n";
        }
        else {
            cout << bel[qry] << "\n";
        }
    }
}

signed main() {
    ios::sync_with_stdio(false);
```

[Meteor Shower S]([P2895 [USACO08FEB\] Meteor Shower S - 洛谷 | 计算机科学教育新生态](https://www.luogu.com.cn/problem/P2895))

```c
#include <bits/stdc++.h>
#define int long long
using namespace std;
const int N = 310;

int m, dist[N][N], fire[N][N];
int dx[] = {1, -1, 0, 0};
int dy[] = {0, 0, 1, -1};
typedef pair<int, int>PII;
queue<PII>q;

int bfs() {
    q.push({0, 0});
    dist[0][0] = 0;
    while(!q.empty()) {
        auto it = q.front();
        q.pop();

        for(int i=0; i<4; i++) {
            int a = it.first + dx[i];
            int b = it.second + dy[i];

            if(a<0 || b<0) continue;
            if(dist[a][b]) continue;
            if(dist[it.first][it.second] +1 >= fire[a][b]) continue;

            dist[a][b] = dist[it.first][it.second] + 1;
            q.push({a, b});

            if(fire[a][b] > 1e9) return dist[a][b];
        }
    }
}

void solve() {
	cin >> m;
    memset(fire, 0x3f, sizeof fire);
    while(m --) {
        int x, y, t;
        cin >> x >> y >> t;
        fire[x][y] = min(t, fire[x][y]);
        for(int i = 0; i < 4; i ++) {
            int a = x + dx[i];
            int b = y + dy[i];
            if(a < 0 || b < 0 || a > 301 || b > 301) continue;
            fire[a][b] = min(t, fire[a][b]);
        } 
    }
    int res = bfs();
    cout << res << endl;
}

signed main() {
    ios::sync_with_stdio(false);
    cin.tie(0);
    cout.tie(0);
    
    solve();
    return 0;
}
```

[汽车拉力比赛]([P2658 汽车拉力比赛 - 洛谷 | 计算机科学教育新生态](https://www.luogu.com.cn/problem/P2658))

```
#include <bits/stdc++.h>
#define int long long
using namespace std;
const int N = 1010;

int n, m, sx, sy, cnt_flag = 0;
int flag[N][N], high[N][N];
bool st[N][N];
typedef pair<int, int> PII;
queue<PII> q;
int dx[] = {1, -1, 0, 0};
int dy[] = {0, 0, 1, -1};

void clear(queue<PII> &q) {
    queue<PII> empty;
    swap(empty, q);
}

bool check(int mid) {
    q.push({sx, sy});
    st[sx][sy] = true;
    int cnt = 1;

    while(!q.empty()) {
        auto it = q.front();
        q.pop();

        for(int i = 0; i < 4; i ++) {
            int a = it.first + dx[i];
            int b = it.second + dy[i];

            if(st[a][b]) continue;
            if(a < 1 || b < 1 || a > n || b > m) continue;
            if(abs(high[a][b] - high[it.first][it.second]) > mid) continue;

            q.push({a, b});
            st[a][b] = true;
            
            if(flag[a][b] == 1) {
                cnt ++;
                if(cnt == cnt_flag) {
                    return true;
                }
            }
        }
    }
    return false;
}

void solve() {
	cin >> n >> m;
    for(int i = 1; i <= n; i ++) {
        for(int j = 1; j <= m; j ++) {
            cin >> high[i][j];
        }
    }

    for(int i = 1; i <= n; i ++) {
        for(int j = 1; j <= m; j ++) {
            cin >> flag[i][j];
            if(flag[i][j] == 1) {
                cnt_flag ++;
            }
        }
    }

    for(int i = 1; i <= n; i ++) {
        for(int j = 1; j <= m; j ++) {
            if(flag[i][j] == 1) {
                sx = i, sy = j;
                break;
            }
        }
    }

    int l = 0, r = 1e9 + 1, ans = -1;
    while(l <= r) {
        int mid = l + r >> 1;
        clear(q);
        memset(st, false, sizeof st);
        if(check(mid)) {
            r = mid - 1;
            ans = mid;
        }
        else l = mid + 1;
    }
    cout << ans << endl;
}

signed main() {
    ios::sync_with_stdio(false);
    cin.tie(0);
    cout.tie(0);
    
    solve();
    return 0;
}
```

[奇怪的电梯(queue + pair模板)]([P1135 奇怪的电梯 - 洛谷 | 计算机科学教育新生态](https://www.luogu.com.cn/problem/P1135))

```
#include <bits/stdc++.h>
#define int long long
using namespace std;
const int N = 410;

int n, a, b, flr[N];
bool flag, st[N];
typedef pair<int, int> PII;
queue<PII> q;

void bfs() {
	q.push({a, 0});
	
	while(!q.empty()) {
		auto it = q.front();
		q.pop();
		
		int f = it.first;
		int ans = it.second;
		
		if(f == b) {
			cout << ans << endl;

			return ;

			st[up] = true;
		}
		if(down > 0 && !st[down]) {
			q.push({down, ans + 1});
			st[down] = true;
		}
	}
	return ;
}

void solve() {
	cin >> n >> a >> b;
	for(int i = 1; i <= n; i ++) cin >> flr[i];
	bfs();
	if(!flag) {
		cout << "-1" << endl;
	}

    cin.tie(0);
    cout.tie(0);
    
    solve();
```

[方格取数]([P1004 [NOIP2000 提高组\] 方格取数 - 洛谷 | 计算机科学教育新生态](https://www.luogu.com.cn/problem/P1004))

```
#include <bits/stdc++.h>
#define int long long
using namespace std;
const int N = 10;

int n, maxn, g[N][N];
int ex, ey, sx = 11, sy = 11;

void dfs(int x, int y, int sum, int tp) {
    if(x == sx && y == sy && tp == 1) {
        maxn = max(maxn, sum);
        return ;
    }
    if(x < sx || y < sy || x > ex || y > ey) return ;
    if(x == ex && y == ey) tp = 1;

    int z = g[x][y];
    g[x][y] = 0;
    if(tp) {
        dfs(x - 1, y, sum + z, tp);
        dfs(x, y - 1, sum + z, tp);
    }
    else {
        dfs(x + 1, y, sum + z, tp);
        dfs(x, y + 1, sum + z, tp);
    }
    g[x][y] = z;
}

void solve() {
    cin >> n;
    int x, y, t;
    while(cin >> x >> y >> t && x) {
        ex = max(ex, x);
        ey = max(ey, y);
        sx = min(sx, x);
        sy = min(sy, y);
        g[x][y] = t;
    }
    dfs(sx, sy, 0, 0);
    cout << maxn << endl;
}

signed main() {
    ios::sync_with_stdio(false);
    cin.tie(0);
    cout.tie(0);
    
    solve();
    return 0;
}
```

```
#include <bits/stdc++.h>
#define int long long
using namespace std;

int n, k, res = 0;
int arr[10];

void dfs(int x, int start, int sum) {
	if(x > k) {
		if(sum == n) res ++;
		return ;
	}

		arr[x] = i;
		dfs(x + 1, i, sum + i);
	}
}

void solve() {
    cin >> n >> k;
    dfs(1, 1, 0);
    cout << res << endl;
}
```

### 记忆化搜索

[摆花(记忆化搜索)]([P1077 [NOIP2012 普及组\] 摆花 - 洛谷 | 计算机科学教育新生态](https://www.luogu.com.cn/problem/P1077))

```
#include <bits/stdc++.h>
#define int long long
using namespace std;
const int N = 110;
const int mod = 1e6 + 7;

int n, m;
int a[N], mem[N][N];

int dfs(int x, int sum) {
	if(sum == m) return 1;
	if(sum > m) return 0;
	if(x > n) return 0;
	if(mem[x][sum]) return mem[x][sum];
	
	int ans = 0;
	for(int i = 0; i <= a[x]; i ++) {
		ans = (ans + dfs(x + 1, sum + i)) % mod;
	}
	mem[x][sum] = ans;
	return ans;
}

void solve() {
    cin >> n >> m;
    for(int i = 1; i <= n; i ++) cin >> a[i];
    int res = dfs(1, 0);
    cout << res << endl;
}

signed main() {
    ios::sync_with_stdio(false);
    cin.tie(0);
    cout.tie(0);
    
    solve();
    return 0;
}
```
