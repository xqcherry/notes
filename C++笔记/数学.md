## 位运算

<font color=salmon>左移和右移</font>

`num << i` 表示将`num`的二进制表示向左移动 2<sup>i</sup> 位所得的值。

`num >> i` 表示将`num`的二进制表示向右移动 2<sup>i</sup>  位所得的值

```c++
p * 2 = p << 1		p / 2 = p >> 1
p * 4 = p << 2		p / 4 = p >> 2
...
int mid = (l + r) / 2  即：
int mid = (l + r) >> 1;
当数据量过大时， 为避免溢出，通常写作
    int mid = l + ((r - l) >> 1);
```



<font color=salmon>按位与， 按位或，异或</font>

`((num >> i) & 1)` 表示取`num`的最低位二进制位

> 将num左移 i 位后与 1 进行与运算

假设结果每一位结果为`x[i]`

可以通过`x << i`表示`2的i次方`来还原每一位后累加

```c++
for(int i = 31; i >= 0; i --) {
	int c = ((x >> i) & 1);
}
```



`x & 1`也可以判断奇偶性 ———— 偶0奇1

```c++
if(x & 1) puts(x为奇数);
else puts(x为偶数)
```



`p<<1 | 1` 会生成一个比 `2 * p` 大 1 的奇数，即 `2*p + 1`



## 高精度

### <font color=DarkSeaGreen>高精度模板</font>

> ##### 高精度加法

```c
// C = A + B, A >= 0, B >= 0
vector<int> add(vector<int> &A, vector<int> &B) {
	if(A.size() < B.size()) return add(B, A);
	
	vector<int> C;
	int t = 0;
	for(int i = 0; i < A.size(); i ++) {
		t += A[i];
		if(i < B.size()) {
			t += B[i];
		}
		C.push_back(t % 10);
		t /= 10;
	}
	if(t) C.push_back(t);
	return C;
}
```

> ##### 高精度减法

```c
// C = A - B, 满足A >= B, A >= 0, B >= 0
vector<int> sub(vector<int> &A, vector<int> &B) {
	vector<int> C;
	for(int i = 0, t = 0; i < A.size(); i ++) {
		t = A[i] - t;
		if(i < B.size()) {
			t -= B[i];
		}
		C.push_back((t + 10) % 10);
		if(t < 0) t = 1;
		else t = 0;
	}
	while(C.size() > 1 && C.back() == 0) C.pop_back();
	return C;
}
```

> ##### 高精度乘法

```c
// C = A * b, A >= 0, b >= 0
vector<int> mul(vector<int> &A, int b) {
    vector<int> C;
    int t = 0;
    for (int i = 0; i < A.size() || t; i ++ ) {
        if (i < A.size()) t += A[i] * b;
        C.push_back(t % 10);
        t /= 10;
    }
    while (C.size() > 1 && C.back() == 0) C.pop_back();
    return C;
}
```

## 数论

### 筛法

#### 埃拉托斯特尼筛法

```
vector<int> primes;
bool is_prime[N];

void sieve(int n) {
	is_prime[0] = is_prime[1] = false;
	for(int i = 2; i <= n; i ++) is_prime[i] = true;
	for(int i = 2; i <= n; i ++) {
		if(is_prime[i]) {
			primes.push_back(i);
			if(i * i > n) continue;
			for(int j = i * i; j <= n; j += i) {
				is_prime[j] = false;
			}
		}
	}
} 
```

#### 线性筛法(欧拉筛法)

埃氏筛法仍有优化空间, 它会将一个合数重复多次标记, 线性筛能让每个合数标记一次

```
vector<int> primes;
bool is_prime[N];

void sieve(int n) {
	for(int i = 2; i <= n; i ++) is_prime[i] = true;
	for(int i = 2; i <= n; i ++) {
		if(is_prime[i]) {
			primes.push_back(i);
		}
		for(int j = 0; j < primes.size(); j ++) {
			int tmp = primes[j];
			if(i * tmp > n) break;
			is_prime[i * tmp] = false;
			if(i % tmp == 0) { // 即i之前被prime[j]筛过
				break;
			}
		}
	}
}
```





## 排列与组合

<font color=salmon>全排列模板</font>

```c
#include <bits/stdc++.h>
#define int long long
using namespace std;
const int N = 50;

int n, a[N];
bool st[N];

void dfs(int x) {
    if(x > n) {
        for(int i = 1; i <= n; i ++) {
            cout << "    " << a[i];
        }
        cout << "\n";
        return ;
    }
    for(int i = 1; i <= n; i ++) {
        if(!st[i]) {
            a[x] = i;
            
            st[i] = true;
            dfs(x + 1);
            st[i] = false;
        }
    }
}

void solve() {    
    cin >> n;
    dfs(1);
}

signed main() {
    ios::sync_with_stdio(false);
    cin.tie(0);
    cout.tie(0);
    
    solve();
    return 0;
}
```

**gcd(a, b) * lcm(a, b) = a * b**

```c
int gcd(int a, int b) {
    while(b) {
        int t = a;
        a = b;
        b = t % b;
    }
    return a;
}

int lcm(int a, int b) {
    return (a / gcd(a, b)) * b;
}
```
