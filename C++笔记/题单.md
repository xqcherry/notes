[滑动窗口](https://www.acwing.com/problem/content/156/)

```c++
int q[N];
int hh = 0, tt = -1;
for(int i = 0; i < n; i ++) {
    while(hh <= tt && i - k + 1 > q[hh]) hh ++;	// 保证队列窗口合法
    while(hh <= tt && a[q[tt]] >= a[i]) tt --;	// 保证窗口内值最小
    
    q[++ tt] = i;
    if(i - k + 1 > 0)
        cout << a[q[hh]] << " " // 队头始终最小
}
```

[多重背包问题 II](https://www.acwing.com/problem/content/5/)(二进制优化O(nmlogS))

```c++
// cnt为分组数，k为倍增数
int cnt = 0;
for(int i = 1; i <= n; i ++) {
    int a, b, s;
    cin >> a >> b >> s;
    
    int k = 1;
    while(k <= s) {
        cnt ++;
        v[cnt] = k * a;
        w[cnt] = k * b;
        s -= k;
        k *= 2;
    }
    if(s) {
         cnt ++;
         v[cnt] = s * a;
         w[cnt] = s * b;
    }
}
// 分组后变成0-1背包
for(int i = 1; i <= cnt; i ++)
    for(int j = m; j >= v[i]; j --)
        f[j] = max(f[j], f[j - v[i]] + w[i]);
```

[多重背包问题 III](https://www.acwing.com/problem/content/6/)(单调队列优化O(nm))

```css
1.按照余数分组(同组内有规律,即遍历每个r从0~v[i])
2.同组内g[q] + (j - q)/v * w
3.维护队列
 - 1.判断队列是否合法(是否超过最大可选容量)
 - 2.比较 g[队列尾部] + 从尾部到当前的价值 与 g[当前]
		 g[q[tt]] + (j - q[tt]) / v[i] * w[i] <= g[j]
4.更新状态(头部始终最优，直接转移)
```

```c++
// 拷贝数组优化至一维
int q[N], f[N], g[N];
for(int i = 1; i <= n; i ++) {
    memcpy(g, f, sizeof g);
    
    for(int r = 0; r < v[i]; r ++) { 	//余数分组
        int hh = 0, tt = -1;
        for(int j = r; j <= m; j += v[i]) { 	// 处理同余类
            while(hh <= tt && j - q[hh] > s[i] * v[i]) hh ++;
            while(hh <= tt && g[q[tt]] + (j - q[tt]) / v[i] * w[i] <= g[j]) tt --;
            
            q[++ tt] = j;
            f[j] = g[q[hh]] + (j - q[hh]) / v[i] * w[i];
        }
    }
    cout << f[m] << "\n";
}
```

```c++
// a: 文本串, b: 模式串
void kmp() {
    // 构建kmp数组
    int j = 0; // 前缀指针
    for(int i = 2; i <= lb; i ++ ) {
        while(J && b[i] != b[j + 1]) j = kmp[j];
        if(b[i] == b[j + 1]) j ++;
        kmp[i] = j;
    }
    // 匹配
    int j = 0;
    for(int i = 1; i <= la; i ++ ) {
        while(j && a[i] != b[j + 1]) j = kmp[j];
        if(a[i] == b[j + 1]) j ++;
        if(j == lb) {
            cout << i - lb + 1 << "\n";
            j = kmp[j];
        }
    }
}
```
