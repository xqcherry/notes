## <font color=salmon>存图方法</font>

### 邻接矩阵

- 适用场景：稠密图、节点数少（通常n ≤ 1000）

- 快速查询边存在性
- 无法处理重边（除非用权值记录次数）

```c
int g[N][N];
// 添加边（有向图）
void add(int u, int v, int w) {
    g[u][v] = w;
}
// 遍历u的邻接点
for(int v = 0; v < n; v ++) {
    if(!g[u][v])
        printf("%%d->%d: %d\n", u, v, g[u][v]);
}
```

-  适用场景：通用存储、支持动态增删边

- 工程常用链表实现，算法题常用vector实现

```c
    cnt = 0;

   	head[u] = cnt ++;	// 更新头节点索引
```

```c
#include <bits/stdc++.h>
#define int long long
using namespace std;

int n, m, t, num = 1;
bool st[5001];
vector<int> adj[5001], tar[5001];
queue<int> q;

void bfs(int x, int cnt, vector<int> adj[], vector<int> tar[]) {
	st[x] = 1;
	q.push(x);
	tar[cnt].push_back(x);
	
	while(!q.empty()) {
		auto it = q.front();
		q.pop();
		
		for(int i = 0; i < adj[it].size(); i ++) {
			int tmp = adj[it][i];
			if(st[tmp]) continue;
			
			st[tmp] = 1;
			q.push(tmp);
			tar[cnt].push_back(tmp);
		}
	}
}

void solve() {
	int x, y;
	cin >> n >> m >> t;
	for(int i = 1; i <= m; i ++) { // 邻接链表存图
		cin >> x >> y;
		adj[x].push_back(y);
		adj[y].push_back(x);
	}
	
	for(int i = 1; i <= n; i ++) { // 构造新图
		if(!st[i]) {
			bfs(i, num ++, adj, tar);
		}
	}
	
	for(int i = 0; i < t; i ++) { // 询问查找
		int flag = 0;
		cin >> x >> y;
		for(int j = 1; j < num; j ++) {
			if(find(tar[j].begin(), tar[j].end(), x) != tar[j].end()) {
				if(find(tar[j].begin(), tar[j].end(), y) != tar[j].end()) {
					flag = 1;
				}
				break;
			}
		}
		if(flag) cout << "Yes" << endl;
		else cout << "No" << endl;
	}
}

signed main() {
    ios::sync_with_stdio(false); 
    cin.tie(0);
    cout.tie(0);
    
    solve();
    return 0;
}
```

## 最短路

### Floyd算法

> 时间复杂度O(n³)，空间复杂度O(n²)

<font color=slamon>多源最短路径问题，适用带负权边但没有负权回路，稠密图</font>

<font color=salmon>模板</font>

```c
for(int k = 1; i <= n; i ++) { 	 	// 中间点
    for(int i = 1; i <= n; i ++) {		// 起点
        for(int j = 1; j <= n; j ++) {  	// 终点
            if(dist[i][k] < INF && dist[k][j] < INF) {
                dist[i][j] = min(dist[i][j], dist[i][k] + dist[k][j]);
            }
        }
    }
}
```

[Floyd](https://www.luogu.com.cn/problem/B3647)

```c
#include <bits/stdc++.h>
#define int long long
using namespace std;
const int N = 110;
const int INF = 2147483647;

int n, m, dist[N][N];

inline int read() {
    int x = 0, f = 1;
    char ch = getchar();
    while (ch < '0' || ch > '9') {
        if (ch == '-') f = -1; // 处理负数
        ch = getchar();
    }
    while (ch >= '0' && ch <= '9') {
        x = x * 10 + ch - '0'; // 将字符转换为数字
        ch = getchar();
    }
    return x * f;
}

void floyd() {
    for(int k = 1; k <= n; k ++) 
        for(int i = 1; i <= n; i ++) 
            for(int j = 1; j <= n; j ++) 
                if(dist[i][k] < INF && dist[k][j] < INF)
                    dist[i][j] = min(dist[i][j], dist[i][k] + dist[k][j]);
}

void solve() {
    int u, v, w;
    n = read(), m = read();
    for(int i = 1; i <= n; i ++)
        for(int j = 1; j <= n; j ++) {
            if(i == j) dist[i][j] = 0;
            else dist[i][j] = INF;
        }

    for(int i = 1; i <= m; i ++) {
        u = read(), v = read(), w = read();
        dist[u][v] = min(dist[u][v], w);
        dist[v][u] = min(dist[v][u], w);
    }
    floyd();
    for(int i = 1; i <= n; i ++) {
        for(int j = 1; j <= n; j ++)
            cout << dist[i][j] << " ";
        cout << "\n";
    }
        
}
```

[Floyd + 状压DP](https://www.lanqiao.cn/problems/1050/learning/?page=1&first_category_id=1&second_category_id=3&difficulty=20&tag_category_id=60)

```C
#include <bits/stdc++.h>
#define int long long
using namespace std;
const int N = 25;
const int MAXN = 1e9;

int n, m;
int x[N], y[N];
double g[N][N], f[1<<20][N];
double ans = MAXN;

double cal(int x1, int y1, int x2, int y2) {
    return 1.0 * sqrt(pow(x1 - x2, 2) + pow(y1 - y2, 2));
}

void floyd() {
    for(int k = 0; k < n; k ++) {
        for(int i = 0; i < n; i ++) {
            for(int j = 0; j < n; j ++) {
                g[i][j] = min(g[i][j], g[i][k] + g[k][j]);
            }
        }
    }
}

void solve() {
    scanf("%lld%lld", &n, &m);
    for(int i = 0; i < n; i ++)
        scanf("%lld%lld", &x[i], &y[i]);
    
    for(int i = 0; i < n; i ++) {
        for(int j = i + 1; j < n; j ++) {
            double p = cal(x[i], y[i], x[j], y[j]);
            if(p <= m) 
                g[i][j] = g[j][i] = p;
            else
                g[i][j] = g[j][i] = MAXN;
        }
    }

    floyd();
    
    for(int st = 0; st < (1<<n); st ++) {
        for(int i = 0; i < n; i ++) {
            f[st][i] = MAXN;
        }
    }
    f[1][0] = 0;
    for(int st = 1; st < (1<<n); st ++) {
        for(int i = 0; i < n; i ++) {
            if((st & (1<<i)) == 0) continue;
            for(int k = 0; k < n; k ++) {
                if((st & (1<<k)) && k != i) {
                    f[st][i] = min(f[st][i], f[st - (1<<i)][k] + g[k][i]);
                }
            }
        }
    }

    for(int i = 0; i < n; i ++) {
        ans = min(ans, f[(1<<n) - 1][i] + g[i][0]);
    }
    printf("%.2f\n", ans);
}

signed main() {
    ios::sync_with_stdio(false);
    cin.tie(0);
    cout.tie(0);
    
    solve();
    return 0;
}
```



### SPFA算法

> 最坏情况下的时间复杂度为 O(nm)，容易被卡

- 支持负权边，不支持负权环

<font color=salmon>模板</font>

```c++
struct edge {
	int to, w, next;
}edge[N];
int head[N], cnt;

int dist[N], vis[N];  // 标记最短，标记访问与否
queue<int> q; // 存放要松弛边的下标

void add(int u, int v, int w) {
    edge[++cnt] = {v, w, head[u]};
    head[u] = cnt;
}

void spfa() {
    for(int i = 1; i <= n; i ++) dist[i] = INF;
    dist[1] = 0;
    vis[1] = 1;
    
    while(!q.empty()) {
        int u = q.front();
        q.pop();
        vis[u] = 0;
        
        for(int i = head[u]; i; i = edges[i].next[i]) {
            int v = edges[i].to;
            int w = edges[i].w;
            if(dist[v] > dist[u] + w) {
                dist[v] = dist[u] + w;
                if(!vis[v]) {
                    q.push(v);
                    vis[v] = 1;
                }
            }
        }
    }
}
```

```c
    head[u] = cnt;

    for(int i = 1; i <= n; i ++) dist[i] = INF;

        for(int i = head[u]; i; i = edges[i].next) {

    if(nums[x]) return nums[x];
    for(int i = head[x]; i; i = edges[i].next) {
        int v = edges[i].to;
        if(dist[x] - 1 == dist[v]) {
            nums[x] = (nums[x] + dfs(v)) % 100003;
        }
    }
    return nums[x];

	n = read(), m = read();
    for(int i = 1; i <= m; i ++) {
        u = read(), v = read();
        add(u, v);
        add(v, u);
```

### Dijkstra算法

> 时间复杂度:暴力O(n²)，优先队列O(mlogm)

<font color=slamon>单源最短路径问题，边权为非负数，稀疏图</font>

<font color=salmon>模板</font>

```c++
struct edge {
    int to;
    int w;
};

struct node {
    int dis;		// dis: 当前距离
    int u;			// u: 节点编号
};

struct cmp {
    bool operator()(const node& a, const node& b) {
        return a.dis > b.dis; 	// 用 > 生成最小堆
    }
};

vector<vector<edge> > g 	// 邻接表存图(g[u]包含u的所有出边)
vector<int> dist, vis; 		// dist存储起点到各点最短路，vis标记是否确认为最短路
priority_queue<node, vector<node>, cmp> q;
const int MAXN = LLONG_MAX;

void dijkstra(int n, int s) {
    g.resize(n + 1); 	// 节点编号1~n
    dist.resize(n + 1);
    vis.resize(n + 1);
    fill(dist.begin(), dist.end(), MAXN);
    fill(vis.begin(), vis.end(), 0);
    dist[s] = 0;
    q.push({0, s});
    
    while(!q.empty()) {
        int u = q.top().u;
        q.pop();
        
        if(vis[u]) continue;
        vis[u] = 1;
        
        for(const auto& ed : g[u]) {
            int to = ed.to, w = ed.w;
            if(dist[to] > dist[u] + w) {
                dist[to] = dist[u] + w;
                q.push({dist[to], to});
            }
        }
    }
}
```

## 最小生成树

### Prim 算法

> 二叉堆：O((n + m)log n)

<font color=slamon>加点法，稠密图</font>

<font color=salmon>模板</font>

```c++
// 链式向前星存图
struct edge {
	int to, w, next;
}edge[N<<1];
int head[N], cnt;

struct node {
    int dis, u;   // u为编号，dis为到该树根最小边权
    bool operator<(const node& x)const {
        return dis > x.dis;
    }
};
int dist[N], vis[N];  	// dist存储每个顶点到生成树的最小边权, vis标记顶点是否已加入生成树
int res, tot;  // res生成树总权重, tot已加入生成树的顶点数
priority_queue<node> q;

void add(int u, int v, int w) {
    edge[++cnt] = {v, w, head[u]};
    head[u] = cnt;
}
// 基本同dijkstra
void prime() {
    for(int i = 1; i <= n; i ++) dist[i] = INF;
    dist[1] = 0;
    q.push({0, 1});
    
    while(!q.empty() && tot != n) {
        int u = q.top().u;
        int dis = q.top().dis;
        q.pop();
        
        if(vis[u]) continue;
        vis[u] = 1;
        tot ++;
        res += dis;
        
        for(int i = head[u]; i; i = edge[i].next) {
            int to = edge[i].to;
            int w = edge[i].w;
            if(dist[to] > w) {
                dist[to] = w;
                q.push({dist[to], to});
            }
        }
    }
}
```

### Kruskal 算法

> 使用O(mlogm)的排序，O(mlogn)的并查集就能得到时间复杂度O(mlogm)的Kruskal算法

<font color=slamon>加边法，稀疏图</font>

- 在Kruskal算法中处理无向图时，**无需双向存储边**
- Kruskal算法关注的是边的**连接关系**而非方向性，无向图中的边 ` u↔v`，在并查集中只需判断一次连通性
- 无论边是存储为 u,v还是 v,u，边的权重 w 在排序时的相对位置不会改变，算法的选择顺序不变
- 不能用链式前向星存，用边列表存

<font color=salmon>模板</font>

```c
struct edge {
    int u, v, w;
} edges[N];
int cnt;
void add(int u, int v, int w) {
    edges[++cnt] = {u, v, w};
}

int pa[N];
int n, m;// n为顶点数，m为边数
int tot, res; // tot记录存储节点个数，res记录总权和

int unfind(int x) {
    if(pa[x] != x) {
        pa[x] = unfind(pa[x]);
        return pa[x];
    }
    return x;
}
void unionset(int x, int y) {
    int rx = unfind(x);
    int ry = unfind(y);
    pa[rx] = ry;
}

bool cmp(const edge &x, const edge &y) {
    return x.w < y.w;
}

void kruskal() {
    sort(edges + 1, edges + 1 + m, cmp);
    for(int i = 1; i <= n; i ++) pa[i] = i;
    for(int i = 1; i <= m && tot < n - 1; i ++) {
        int u = edges[i].u;
        int v = edges[i].v;
        int w = edges[i].w;
        if(unfind(u) != unfind(v)) {
            unionset(u, v);
            res += w;
            tot ++;
        }
    }
}
```
